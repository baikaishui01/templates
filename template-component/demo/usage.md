本 Demo 演示一行文字的用法。

```jsx
import React, { Component } from 'react';
import ReactDOM from 'react-dom';
import {FileName} from 'packageName';

class App extends Component {
  render() {
    return (
      <div>
        <FileName title="FileName demo" />
      </div>
    );
  }
}

ReactDOM.render((
  <App />
), mountNode);
```
